	public function page($total, $p){
        $pageNum = 5;
        $p = $p < 1 ? 1 : $p;
        $p = $p > $total ? $total : $p;
        $total = $total < $p ? $p : $total;

        // 计算开始页
        $start = $p - floor($pageNum / 2);
        $start = $start < 1 ? 1 : $start;

        // 计算结束页
        $end = $p + floor($pageNum /2);
        $end = $p >= $total ? $total : $end;

        $current = $end - $start + 1;
        // 左调整
        if ($current < $pageNum && $start > 1){
            $start = $start - ($pageNum-$current);
            $start = $start<1 ? 1 : $start;
            $current = $end-$start+1;
        }
        // 右边调整
        if($current<$pageNum && $end<$total){
            $end = $end + ($pageNum-$current);
            $end = $end>$total? $total : $end;
        }

        $output = '';
        if ($p > 1){
            $output .= '<a href="/tv/bang/tv_list.php?p=1" title="第一页"><<</a>';
        }
        for ($i = $start; $i <= $end; $i++){
            if ($i == $p){
                $output .= '<a href="javascript:void(0)" class="current">'.$i.'</a>';
            }else{
                $output .= '<a href="/tv/bang/tv_list.php?p='.$i.'">'.$i.'</a>';
            }
        }

        if ($p < $end){
            $output .= '<a href="/tv/bang/tv_list.php?p='.$total.'" title="第一页">>></a>';
        }
        $this->output['pages'] = $output;

    }
